DROP DATABASE IF EXISTS zay_shop;
create DATABASE zay_shop;
use zay_shop;

create table users(
    name varchar(45) not null,
    email varchar(45) not null primary key
);
create table posts(
    id int AUTO_INCREMENT primary key,
    name varchar(45) not null,
    email varchar(45) not null,
    subject varchar(75) not null,
    message TEXT(500) not null
    );
create table admins(
    id int primary key,
    login varchar(45) not null,
    passwords varchar(75) not null
 );
